Virus(Init){
  *(Init):
  | (Cell(V,N) & Closest(@,N)) ? Pop(N) :(Init)
  | (Cell(V,S) & Closest(@,S)) ? Pop(S) :(Init)
  | (Cell(V,O) & Closest(@,O)) ? Pop(O) :(Init)
  | (Cell(V,E) & Closest(@,E)) ? Pop(E) :(Init)
  | (Cell(V,N) & Closest(P,N)) ? Wizz(N) :(Init)
  | (Cell(V,S) & Closest(P,S)) ? Wizz(S) :(Init)
  | (Cell(V,O) & Closest(P,O)) ? Wizz(O) :(Init)
  | (Cell(V,E) & Closest(P,E)) ? Wizz(E) :(Init)
}
